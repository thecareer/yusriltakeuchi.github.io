options:
	PJ: &a&lPenjual&7: &f
	
on join:
	set {npc.%player%.pj} to false
	set {clear.%player%.pj} to true
	
every 10 seconds:
	if text from "http://yusriltakeuchi.com/version.txt" is not "V1.1":
		broadcast "Updating [Taya Shop]"
		download from "http://yusriltakeuchi.com/tayashop.txt" to "plugins/Skript/scripts/tayashop.sk"
		wait 5 ticks
		execute console command "/skript reload tayashop"
		broadcast "Finished Updating [tayashop]]"
	
on right click on villager:
	if name of clicked villager is "&a&lPenjual":
		if {npc.%player%.pj} is false:
			show mob spawner flames at the clicked entities to the player
			execute console command "/playsound random.orb %player%"
			wait 10 ticks
			if {clear.%player%.pj} is true:
				message "{@PJ}Hai, aku adalah seorang &a&lPenjual"
				wait 50 ticks
				message "{@PJ}Terima kasih telah datang ke toko ku dan membeli sesuatu"
				wait 50 ticks
				message "{@PJ}Silahkan di pilih-pilih"
				set {clear.%player%.pj} to false
				show "s;1;i;apple;n;&eApple;l;&5*Price $30;c;sudo %player% tayashop buy apple;s;2;i;bread;n;&eBread;l;&5*Price $45;c;sudo %player% tayashop buy bread;s;3;i;raw porkchop;n;&eRaw Porkchop;l;&5*Price $20;c;sudo %player% tayashop buy rp;s;4;i;steak;n;&eSteak;l;&5*Price $45;c;sudo %player% tayashop buy steak;s;5;i;goldenapple;n;&eGolden Apple;l;&5*Price $120;c;sudo %player% tayashop buy goldenapple" with 4 rows named "&a&lTaya&f&lShop" to player
				stop
			else:
				show "s;1;i;apple;n;&eApple;l;&5*Price $30;c;sudo %player% tayashop buy apple;s;2;i;bread;n;&eBread;l;&5*Price $45;c;sudo %player% tayashop buy bread;s;3;i;raw porkchop;n;&eRaw Porkchop;l;&5*Price $20;c;sudo %player% tayashop buy rp;s;4;i;steak;n;&eSteak;l;&5*Price $45;c;sudo %player% tayashop buy steak;s;5;i;goldenapple;n;&eGolden Apple;l;&5*PRice $120;c;sudo %player% tayashop buy goldenapple" with 4 rows named "&a&lTaya&f&lShop" to player
				stop
			
			
command /tayashop [<text>] [<text>]:
	trigger:
		if argument 1 is "buy":
			if argument 2 is "apple":
				if player's money is above 30:
					execute console command "/playsound random.orb %player%"
					execute console command "/give %player% apple 1"
					message "{@PJ}$30 telah diambil dari akunmu!"
					remove 30 from the player's account
					stop
				else:
					message "{@PJ}&cMaaf, uangmu tidak mencukupi!"
					stop
			if argument 2 is "bread":
				if player's money is above 45:
					execute console command "/playsound random.orb %player%"
					execute console command "/give %player% bread 1"
					message "{@PJ}$45 telah diambil dari akunmu!"
					remove 45 from the player's account
					stop
				else:
					message "{@PJ}&cMaaf, uangmu tidak mencukupi!"
					stop
			if argument 2 is "rp":
				if player's money is above 20:
					execute console command "/playsound random.orb %player%"
					execute console command "/give %player% 319 1"
					message "{@PJ}$20 telah diambil dari akunmu!"
					remove 20 from the player's account
					stop
				else:
					message "{@PJ}&cMaaf, uangmu tidak mencukupi!"
					stop
			if argument 2 is "steak":
				if player's money is above 45:
					execute console command "/playsound random.orb %player%"
					execute console command "/give %player% steak 1"
					message "{@PJ}$45 telah diambil dari akunmu!"
					remove 45 from the player's account
					stop
				else:
					message "{@PJ}&cMaaf, uangmu tidak mencukupi!"
					stop
			if argument 2 is "goldenapple":
				if player's money is above 120:
					execute console command "/playsound %player% goldenapple"
					execute console command "/give %player% goldenapple 1"
					message "{@PJ}$120 telah diambil dari akunmu!"
					remove 120 from the player's account
					stop
				else:
					message "{@PJ}&cMaaf, uangmu tidak mencukupi!"
					stop
